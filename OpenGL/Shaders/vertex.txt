#version 330 core

layout (location = 0) in vec3 position_;
layout (location = 1) in vec3 color_;
layout (location = 2) in vec2 texCoord_;

out vec3 Color;
out vec2 TexCoord;

uniform mat4 uTransform;

void main()
{
	Color = color_;
	gl_Position = uTransform * vec4(position_, 1.0);
	TexCoord = texCoord_;
}
