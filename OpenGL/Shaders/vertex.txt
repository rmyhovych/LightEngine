#version 330 core

layout (location = 0) in vec3 position_;
layout (location = 1) in vec3 normal_;

out vec3 Normal;
out vec3 FragPos;

uniform mat4 uRotation = mat4(1.0);
uniform mat4 uTranslation;
uniform mat4 uView;
uniform mat4 uProjection;

void main()
{
	mat4 model = uTranslation * uRotation;

	Normal = vec3(normalize(uRotation * vec4(normal_, 1.0)));
	FragPos = vec3(model * vec4(position_, 1.0));

	gl_Position = uProjection * uView * model * vec4(position_, 1.0);
}
