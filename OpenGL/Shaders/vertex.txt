#version 330 core

layout (location = 0) in vec3 position_;
//layout (location = 1) in vec3 normal_;

out vec3 Color;

uniform vec3 uColor;
uniform mat4 uModel;
uniform mat4 uView;
uniform mat4 uProjection;

void main()
{
	gl_Position = uProjection * uView * uModel * vec4(position_, 1.0);
	Color = uColor;
}
