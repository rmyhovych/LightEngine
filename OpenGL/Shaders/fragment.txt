#version 330 core

in vec3 Normal;
in vec3 FragPos;

out vec4 FragColor;

uniform vec3 uColor;
uniform vec3 uLightColor;
uniform vec3 uLightPos;

void main()
{
	float ambient = 0.1;

	float diffuse = max(dot(Normal, normalize(uLightPos - FragPos)), 0.0);

	vec3 result = (ambient + diffuse) * uLightColor * uColor;
	FragColor = vec4(result, 1.0f);
}
