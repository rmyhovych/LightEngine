#version 330 core

in vec3 Normal;
in vec3 FragPos;

out vec4 FragColor;

uniform vec3 uColor;
uniform vec3 uLightColor;
uniform vec3 uLightPos;
uniform vec3 uViewPos;

void main()
{
	vec3 lightDir = uLightPos - FragPos;
	float fraction = 1/(1 + dot(lightDir, lightDir) / 100);
	lightDir = normalize(lightDir);

	vec3 reflectDir = reflect(-lightDir, Normal);
	vec3 viewDir = normalize(uViewPos - FragPos);

	float ambient = 0.1;
	float diffuse = fraction * 1.0 * max(dot(Normal, lightDir), 0.0);
	float specular = fraction * 0.5 * pow(max(dot(viewDir, reflectDir), 0.0), 32);

	vec3 result = (ambient + diffuse + specular) * uLightColor * uColor;
	FragColor = vec4(result, 1.0f);
}
